using System;
class Program
{
    public static void Main()
    {
        
            int temp;
            int a = 10;
            int b = 20;
            Console.WriteLine("Before swap {0} {1}", a, b);

            //swapping logic
            temp= a;
            a = b;
            b = temp;
        Console.WriteLine("After swap {0} {1}", a, b);
        
    }
}
